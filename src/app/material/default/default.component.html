Install Angular Material

ng add @angular/material

Create a shared module to share Angular Material dependencies

    import { NgModule } from '@angular/core';
    import { MatButtonModule } from '@angular/material/button';
    import { MatIconModule } from '@angular/material/icon';
    import { MatMenuModule } from '@angular/material/menu';

    ....

    @NgModule({
        declarations: [...],
        imports: [MatButtonModule, MatIconModule, MatMenuModule],
        exports: [MatButtonModule, MatIconModule, MatMenuModule]
    })

    export class MaterialcompModule {}

Use this module in your app.module.ts or any other module

    import { BrowserModule } from '@angular/platform-browser';
    import { NgModule } from '@angular/core';
    ....
    import { MaterialcompModule } from './shared/MaterialcompModule.module';
    ....

    @NgModule({
    declarations: [....],
    imports: [
        ...
        MaterialcompModule,
        ...
    ],
    providers: [...],
    bootstrap: [AppComponent]
    })
    export class AppModule { }
